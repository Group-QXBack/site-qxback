-- Tabela para armazenar informações da empresa indicada
CREATE TABLE indicacoes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome_empresa VARCHAR(255) NOT NULL,
    cnpj VARCHAR(18) NOT NULL,
    cpf VARCHAR(14) NOT NULL,
    telefone_empresa VARCHAR(15),
    celular_empresa VARCHAR(15) NOT NULL,
    email_empresa VARCHAR(255) NOT NULL,
    data_indicacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tabela para armazenar informações do contato associado à indicação
CREATE TABLE contatos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    indicacao_id INT NOT NULL,
    nome_contato VARCHAR(255) NOT NULL,
    cargo_contato VARCHAR(255),
    celular_contato VARCHAR(15) NOT NULL,
    email_contato VARCHAR(255) NOT NULL,
    FOREIGN KEY (indicacao_id) REFERENCES indicacoes(id)
);

-- Tabela para armazenar relação entre indicações e áreas
CREATE TABLE indicacoes_areas (
    indicacao_id INT,
    area_id INT,
    PRIMARY KEY (indicacao_id, area_id),
    FOREIGN KEY (indicacao_id) REFERENCES indicacoes(id),
    FOREIGN KEY (area_id) REFERENCES areas(id)
);
